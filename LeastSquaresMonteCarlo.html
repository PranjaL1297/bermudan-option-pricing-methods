def bs_price(S0, K, r, T, sigma, option = 'call'):
    
    #S: spot price
    #K: strike price
    #r: interest rate
    #T: time to maturity
    #sigma: volatility of underlying asset
    
    d1 = (np.log(S0/K) + (r+0.5*sigma**2)*T)/(sigma*np.sqrt(T))
    d2 = (np.log(S0/K) + (r-0.5*sigma**2)*T)/(sigma*np.sqrt(T))
    
    if option == 'call':
        result = (S0*si.norm.cdf(d1, 0.0, 1.0) - K*np.exp(-r*T)*si.norm.cdf(d2, 0.0, 1.0))
    if option == 'put':
        result = (K*np.exp(-r*T)*si.norm.cdf(-d2, 0.0, 1.0) - S0*si.norm.cdf(-d1, 0.0, 1.0))        
    return result

def LSMC_option_price(S0, K, r, T, sigma, N, Nsim):
    dt = T/N  # time interval
    df = np.exp(-r*dt)  # discount factor per time time interval
    
    # Simulation of asset price
    S = np.zeros((Nsim, N+1))  # stock price matrix
    S[:, 0] = S0  # intial values for stock price
    for j in range(1, N+1):
        eps = np.random.normal(0,1, int(Nsim/2))
        eps = np.concatenate((eps, -eps))  # antithetic variates
        eps = eps - np.mean(eps)  # correct first moment
        eps = eps / np.std(eps)   # correct second moment
        S[:,j] = S[:,j-1]*np.exp((r-sigma**2/2)*dt+sigma*eps*np.sqrt(dt))
    
    h = np.maximum(K - S, 0)  # inner values for put option
    V = np.zeros((Nsim,N+1))  # value matrix
    V[:,-1] = h[:,-1]
    
    # Valuation by LSM
    for j in range(N-1, 0, -1):
        Idx = np.where(V[:,j + 1]>0)[0]                        # Choose in the money paths
        rg = np.polyfit(S[Idx,j], V[Idx,j+1]*df, 3)            # polynomial regression
        C = np.polyval(rg, S[:,j])                             # evaluation of polynomial regression
        V[:,j] = np.where(h[:,j] > C, h[:,j],  V[:,j + 1]*df)  # exercise decision/optimization
    
    V0 = np.sum(V[:,1]*df)/Nsim  # LSM estimator
    return V0
